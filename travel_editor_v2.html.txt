<script>
// ── 状態 ──────────────────────────────────────────────────────────────────
const S = { media:[], tl:[], bgm:null, idx:0 };
let mid=0, cid=0;

// ── ページ切り替え ────────────────────────────────────────────────────────
function switchPage(name) {
  ['media','settings'].forEach(n => {
    document.getElementById('page-'+n).classList.toggle('active', n===name);
    document.getElementById('nav-'+n).classList.toggle('active', n===name);
  });
  document.getElementById('contentArea').scrollTop = 0;
}

// ── ファイル読み込み（FileReader で base64）────────────────────────────────
function onFiles(files) {
  let n = 0;
  Array.from(files).forEach(f => {
    if (!f.type.match(/^(image|video)\//)) return;
    const isImg = f.type.startsWith('image/');
    const id = ++mid;
    const item = { id, name:f.name, type: isImg?'photo':'video', src:'', duration:0 };
    S.media.push(item);
    n++;

    // FileReader で安全に読み込む
    const fr = new FileReader();
    fr.onload = e => {
      item.src = e.target.result;
      if (!isImg) {
        // 動画の長さを取得
        const v = document.createElement('video');
        v.onloadedmetadata = () => { item.duration = v.duration; renderMedia(); };
        v.src = item.src;
      }
      renderMedia();
    };
    fr.readAsDataURL(f);
  });
  if (n) toast(n+'件追加 🎉','ok');
}

function onBGM(files) {
  if (!files[0]) return;
  S.bgm = files[0];
  document.getElementById('bgmok').style.display='block';
  document.getElementById('bgmok').textContent='🎵 '+files[0].name;
  document.getElementById('bgmvol').style.display='block';
  toast('BGM設定 🎵','ok');
}

// ── 素材グリッド描画 ───────────────────────────────────────────────────────
function renderMedia() {
  const sc = document.getElementById('mscroll');
  sc.innerHTML = '';
  S.media.forEach(item => {
    const d = document.createElement('div');
    d.className = 'mthumb';
    if (item.src) {
      if (item.type==='photo') {
        const img = document.createElement('img');
        img.src = item.src;
        d.appendChild(img);
      } else {
        const v = document.createElement('video');
        v.src = item.src;
        v.preload = 'metadata';
        d.appendChild(v);
      }
    }
    const badge = document.createElement('span');
    badge.className = 'tbadge '+item.type;
    badge.textContent = item.type==='photo'?'PHO':'VID';
    d.appendChild(badge);
    const rm = document.createElement('button');
    rm.className='trm'; rm.textContent='✕';
    rm.onclick = e => { e.stopPropagation(); rmMedia(item.id); };
    d.appendChild(rm);
    if (item.duration) {
      const dur = document.createElement('span');
      dur.className='tdur'; dur.textContent=fmtDur(item.duration);
      d.appendChild(dur);
    }
    d.onclick = () => previewItem(item);
    sc.appendChild(d);
  });
}

function rmMedia(id) {
  S.media = S.media.filter(m=>m.id!==id);
  S.tl = S.tl.filter(c=>c.mediaId!==id);
  renderMedia(); renderTL();
}

// ── プレビュー ─────────────────────────────────────────────────────────────
function previewItem(item) {
  document.getElementById('prevEmpty').style.display='none';
  document.getElementById('prevTitle').style.display='none';
  document.getElementById('prevWm').style.display='block';
  document.getElementById('prevWm').textContent=document.getElementById('inWm').value||'✈ Travel Vlog';
  if (item.type==='photo') {
    document.getElementById('prevImg').src=item.src;
    document.getElementById('prevImg').style.display='block';
    document.getElementById('prevVid').style.display='none';
  } else {
    document.getElementById('prevVid').src=item.src;
    document.getElementById('prevVid').style.display='block';
    document.getElementById('prevImg').style.display='none';
  }
}

function previewClip(clip) {
  if (clip.kind==='title') {
    document.getElementById('prevEmpty').style.display='none';
    document.getElementById('prevImg').style.display='none';
    document.getElementById('prevVid').style.display='none';
    document.getElementById('prevWm').style.display='none';
    document.getElementById('prevTitleTxt').textContent=clip.label;
    document.getElementById('prevTitle').style.display='flex';
    return;
  }
  const item = S.media.find(m=>m.id===clip.mediaId);
  if (item && item.src) previewItem(item);
}

function navClip(dir) {
  if (!S.tl.length) return;
  S.idx = (S.idx+dir+S.tl.length)%S.tl.length;
  previewClip(S.tl[S.idx]);
  hlClip(S.tl[S.idx].id);
}

// ── タイムライン ───────────────────────────────────────────────────────────
function buildTL() {
  if (!S.media.length) { toast('素材を先に追加！','err'); return; }
  S.tl = [];
  const title = document.getElementById('inTitle').value||'My Travel Vlog';
  const sub   = document.getElementById('inSub').value;
  S.tl.push({ id:++cid, kind:'title', label: title+(sub?' / '+sub:''), mediaId:null });
  const sl = parseFloat(document.getElementById('inSl').value)||3;
  S.media.filter(m=>m.type==='photo').forEach(p=>S.tl.push({id:++cid,kind:'photo',mediaId:p.id,label:p.name,slideLen:sl}));
  S.media.filter(m=>m.type==='video').forEach(v=>S.tl.push({id:++cid,kind:'video',mediaId:v.id,label:v.name,trimStart:0,trimEnd:v.duration}));
  renderTL();
  document.getElementById('expBtn').disabled=false;
  toast('タイムライン生成 ✅','ok');
  if (S.tl.length) { S.idx=0; previewClip(S.tl[0]); }
}

function renderTL() {
  const sc=document.getElementById('tlscroll'), em=document.getElementById('tlempty');
  document.getElementById('ccount').textContent=S.tl.length+' クリップ';
  if (!S.tl.length) {
    sc.innerHTML=''; sc.appendChild(em);
    document.getElementById('expBtn').disabled=true; return;
  }
  if (em.parentNode===sc) sc.removeChild(em);
  sc.innerHTML='';
  S.tl.forEach(clip=>{
    const d=document.createElement('div');
    d.className='tlclip clip-'+(clip.kind==='title'?'title':clip.kind);
    d.dataset.cid=clip.id;
    const icon={title:'🅣',photo:'📷',video:'🎥'}[clip.kind]||'●';
    const nm=clip.label.length>12?clip.label.slice(0,12)+'…':clip.label;
    const span1=document.createElement('span'); span1.textContent=icon;
    const span2=document.createElement('span'); span2.textContent=nm;
    const rm=document.createElement('button'); rm.className='tlrm'; rm.textContent='✕';
    rm.onclick=e=>{e.stopPropagation();rmClip(clip.id);};
    d.append(span1,span2,rm);
    d.style.width=clip.kind==='video'?Math.max(72,Math.min((clip.trimEnd||10)*10,160))+'px':clip.kind==='title'?'80px':'64px';
    d.onclick=()=>{openSheet(clip);hlClip(clip.id);previewClip(clip);};
    sc.appendChild(d);
  });
}

function hlClip(id) {
  document.querySelectorAll('.tlclip').forEach(el=>el.classList.toggle('act',+el.dataset.cid===id));
}
function rmClip(id) {
  S.tl=S.tl.filter(c=>c.id!==id); renderTL();
}
function addTitle() {
  S.tl.push({id:++cid,kind:'title',label:'Chapter',mediaId:null}); renderTL(); toast('タイトル追加','ok');
}

// ── クリップシート ─────────────────────────────────────────────────────────
function openSheet(clip) {
  const item=clip.mediaId?S.media.find(m=>m.id===clip.mediaId):null;
  document.getElementById('stitle').textContent=clip.kind==='title'?'📝 タイトル':clip.kind==='photo'?'📷 写真':'🎥 動画';
  let h='';
  if (clip.kind==='title') {
    h=`<div class="field"><label>テキスト</label><input type="text" value="${clip.label.replace(/"/g,'&quot;')}" oninput="updLabel(${clip.id},this.value)"></div>`;
  } else if (clip.kind==='photo') {
    h=`<div class="field"><label>表示時間(秒)</label><input type="number" min="0.5" step="0.5" value="${clip.slideLen||3}" oninput="updSlide(${clip.id},+this.value)"></div>`;
  } else {
    h=`<div class="field"><label>開始(秒)</label><input type="number" min="0" step="0.5" value="${clip.trimStart||0}" oninput="updTrim(${clip.id},'s',+this.value);renderTL()"></div>
       <div class="field"><label>終了(秒)</label><input type="number" min="0" step="0.5" value="${clip.trimEnd||item?.duration||0}" oninput="updTrim(${clip.id},'e',+this.value);renderTL()"></div>
       <div style="font-size:.78rem;color:var(--muted);margin-bottom:14px">元の長さ: ${item?.duration?fmtDur(item.duration):'-'}</div>`;
  }
  h+=`<div style="display:flex;gap:8px;margin-top:4px">
    <button class="btn btn-o" style="font-size:.75rem" onclick="moveClip(${clip.id},-1);renderTL()">← 前へ</button>
    <button class="btn btn-o" style="font-size:.75rem" onclick="moveClip(${clip.id},1);renderTL()">後へ →</button>
  </div>
  <button class="btn btn-d" style="margin-top:10px;font-size:.75rem" onclick="rmClip(${clip.id});closeSheet()">削除</button>`;
  document.getElementById('sbody').innerHTML=h;
  document.getElementById('sheetOv').classList.add('open');
}
function closeSheet(e) {
  if (e && e.target!==document.getElementById('sheetOv')) return;
  document.getElementById('sheetOv').classList.remove('open');
}
function updLabel(id,v){const c=S.tl.find(c=>c.id===id);if(c){c.label=v;renderTL();}}
function updSlide(id,v){const c=S.tl.find(c=>c.id===id);if(c)c.slideLen=v;}
function updTrim(id,k,v){const c=S.tl.find(c=>c.id===id);if(c){if(k==='s')c.trimStart=v;else c.trimEnd=v;}}
function moveClip(id,dir){
  const i=S.tl.findIndex(c=>c.id===id); if(i<0)return;
  const ni=i+dir; if(ni<0||ni>=S.tl.length)return;
  [S.tl[i],S.tl[ni]]=[S.tl[ni],S.tl[i]]; renderTL();
}

function exportProject() {
  if (!S.tl.length){toast('タイムラインが空です','err');return;}
  const title=document.getElementById('inTitle').value||'My Travel Vlog';
  const sub=document.getElementById('inSub').value;
  const wm=document.getElementById('inWm').value||'✈ Travel Vlog';
  const vol=document.getElementById('inVol').value/100;
  const sl=document.getElementById('inSl').value;
  const tr=document.getElementById('inTr').value;
  let cmd=`python3 travel_editor.py ./media/ output.mp4 --title "${title}"`;
  if(sub) cmd+=` --subtitle "${sub}"`;
  if(S.bgm) cmd+=` --bgm "${S.bgm.name}" --bgm-vol ${vol}`;
  cmd+=` --watermark "${wm}"`;
  const data={
    title,subtitle:sub,watermark:wm,
    bgm:S.bgm?{name:S.bgm.name,volume:vol}:null,
    slideDuration:+sl,transition:tr,
    clips:S.tl.map(c=>{
      const item=c.mediaId?S.media.find(m=>m.id===c.mediaId):null;
      return{kind:c.kind,label:c.label,file:item?.name||null,trimStart:c.trimStart,trimEnd:c.trimEnd,slideLen:c.slideLen};
    }),
    pythonCommand:cmd,
    exportedAt:new Date().toISOString()
  };
  const a=document.createElement('a');
  a.href=URL.createObjectURL(new Blob([JSON.stringify(data,null,2)],{type:'application/json'}));
  a.download='travel_project.json';
  a.click();
  toast('書き出し完了 📦','ok');
}

function fmtDur(s){const m=Math.floor(s/60),sc=Math.floor(s%60);return`${m}:${String(sc).padStart(2,'0')}`;}

let toastT;
function toast(msg,type=''){
  const el=document.getElementById('toast');
  el.textContent=msg; el.className=`toast show ${type}`;
  clearTimeout(toastT); toastT=setTimeout(()=>el.classList.remove('show'),2800);
}
